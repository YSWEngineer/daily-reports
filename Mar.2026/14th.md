## 模範解答
```ruby
#!/usr/bin/env ruby
# frozen_string_literal: true

require 'optparse'
require 'etc'

OPTIONS = ARGV.getopts('a', 'r', 'l')
FILETYPES_LIST = {
  'file' => '-',
  'directory' => 'd',
  'link' => 'l'
}.freeze

PERMISSION_LIST = {
  '0' => '---',
  '1' => '--x',
  '2' => '-w-',
  '3' => '-WX',
  '4' => 'r--',
  '5' => 'r-x',
  '6' => 'rw-',
  '7' => 'rwx'
}.freeze

def main
  file_list = make_list
  return if file_list.size.zero?

  OPTIONS['l'] ? file_details : display(file_list)
end

def make_list
  list = Dir.glob('*', OPTIONS['a'] ? File::FNM_DOTMATCH : 0).sort
  OPTIONS['r'] ? list.reverse : list
end

def file_stats
  make_list.map { |list| File.lstat(list) }
end

def total
  total_number = 0
  file_stats.each do |file_element|
    total_number += file_element.blocks
  end
  puts "total #{total_number}"
end

def ftype_permission
  file_stats.map do |f|
    [FILETYPES_LIST[f.ftype], f.mode.to_s(8).slice(-3..-1).chars.map { |char| PERMISSION_LIST[char] }].join
  end
end

def hardlink
  link_numbers = file_stats.map { |file_element| file_element.nlink.to_s }
  max_digit = link_numbers.map(&:size).max
  link_numbers.map { |h| " #{h.rjust(max_digit)}" }
end

def orner_name
  file_stats.map { |file_element| Etc.getpwuid(file_element.uid).name }
end

def group_name
  file_stats.map { |file_element| " #{Etc.getgrgid(file_element.gid).name}" }
end

def filesize
  file_size = file_stats.map { |file_element| file_element.size.to_s }
  longest_filesize = file_size.map(&:size).max
  file_size.map { |f| " #{f.rjust(longest_filesize)}" }
end

def time_stamp
  file_stats.map { |file_element| file_element.mtime.strftime('%_m %_d %R') }
end

def symlink_name
  make_list.map { |list_element| FileTest.symlink?(list_element) ? "#{list_element} -> #{File.readlink(list_element)}" : list_element }
end

def file_details
  total
  result = [ftype_permission, hardlink, orner_name, group_name, filesize, time_stamp, symlink_name]
  result.transpose.map { |display| puts display.join(' ') }
end

def display(file_list)
  display_width = `tput cols`.to_i

  max_columns = 3
  longest_name_size = file_list.map(&:size).max
  minus_columns = (0...max_columns).find { longest_name_size < display_width / (max_columns - _1) } || max_columns - 1
  columns_number = max_columns - minus_columns

  columns_width = display_width / columns_number

  vertical = (file_list.size / columns_number.to_f).ceil

  slice = file_list.map { |d| d.ljust(columns_width) }.each_slice(vertical).to_a

  slice.map { |element| element.values_at(0...vertical) }.transpose.each { |display| puts display.join('') }
end

main
```

### 15. file_detailsメソッド
```ruby
def file_details
  total
  result = [ftype_permission, hardlink, orner_name, group_name, filesize, time_stamp, symlink_name]
  result.transpose.map { |display| puts display.join(' ') }
end
```
`-l`表示全体を担当しているメソッド。
```ruby
# 各メソッドはこのような配列を返している
ftype_permission #=> ['-rw-r--r--', '-rw-r--r--', ...]
hardlink                    #=> [' 1', ' 1', ...]
owner_name          #=> ['yoshiwo', 'yoshiwo', ...]

result = [
  ['-rw-r--r--', '-rw-r--r--'],  # ftype_permission
  [' 1', ' 1'],  # hardlink
  ['yoshiwo', 'yoshiwo']  # owner_name
]

result.transpose
# 行と列を入れ替える
# [
#    ['-rw-r--r--', ' 1', 'yoshiwo'],  # 1ファイル目の全フィールド
#    ['-rw-r--r--', ' 1', 'yoshiwo'],  # 2ファイル目の全フィールド
#  ]

#=> '-rw-r--r--   1  yoshiwo  ...'
```
`transpose`：「行と列を入れ替える」メソッド。今回のコードでは、`transpose`前は**フィールドごとにまとまっていて**、`transpose`後は**ファイルごとにまとまっている**。


## 自分のコードと模範解答の違い
### 1. オプション解析の方法
自分のコード：`OptionParser`を使い、オプションごとに変数（`show_hidden`など）を用意した。
模範解答：`ARGV.getopts`を使い、結果を定数`OPTIONS`のハッシュ一つで管理した。

### 2. プログラムの構造
自分のコード：ファイル末尾の`if/else`で処理を分岐させた。
模範解答：`main`メソッドを定義してプログラムの流れを一箇所にまとめ、最後に`main`を呼び出す形にしている。

### 3. -l表示のフィールド管理
自分のコード：全フィールドをハッシュの配列（`file_metadata`）にまとめて管理し、`widths`で各フィールドの最大幅を計算した。
模範解答：フィールドごとに独立したメソッド（`hardlink`・`owner_name`・`filesize`など）に分け、最後に`transpose`メソッドで1ファイルずつまとめて表示した。

### 4. ファイルタイプの対応範囲
自分のコード：`FTYPE_TO_CHAR`で7種類のファイルタイプに対応した。
模範解答：`FILETYPES_LIST`で3種類（file・directory・link）のみに対応した。

### 5. シンボリックリンクの表示
自分のコード：対応していない。
模範解答：`symlink_name`メソッドでシンボリックリンクの場合に`リンク名 -> 本物のパス`という形式で表示した。

## まとめ
自分のコードはハッシュで全フィールドを一括管理するアプローチ、模範解答はフィールドごとにメソッドに分けて`transpose`メソッドで結合するアプローチという違いがあった。
